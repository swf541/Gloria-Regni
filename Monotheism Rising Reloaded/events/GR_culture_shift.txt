namespace = GRC
# Hispano-Roman Province turns to Andalusian
province_event = {
	id = GRC.1
	desc = EVTDESC_GRC_1
	picture = GFX_evt_stone_church

	trigger = {
		culture = hispanic
		owner = {
			culture_group = arabic 
		}
		NOT = {
			owner = { 
				top_liege = { 
					culture = hispanic 
				} 
			}
		}
	}
	
	mean_time_to_happen = {
		months = 3600
		modifier = {
			factor = 0.5
			religion_group = muslim
		}
		#modifier = {
		#	factor = 0.5
		#	owner = {
		#		culture_group = latin
		#	}
		#}
	}

	option = {
		name = EVTOPTA_GRC_1
		culture = andalusian_arabic
	}
}
