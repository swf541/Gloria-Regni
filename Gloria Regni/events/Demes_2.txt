# Drunkard (Guest)
character_event = {
	id = DEMES.720
	desc = EVTDESC_DEMES_650
	picture = GFX_evt_drunk
	border = GFX_event_normal_frame_religion
	
	hide_from = yes
	is_triggered_only = yes
	
	trigger = {
		has_dlc = "The Old Gods"
		has_character_flag = attending_chariot_race
		
		liege = { 
			has_character_flag = attending_chariot_race
			
		}
		trait = drunkard
	}
	
	option = {
		name = EVTOPTA_TOG_650
		tooltip_info = drunkard
		set_character_flag = feast_drunkard
		hidden_tooltip = {
			liege = {
				set_character_flag = feast_drunkard
				character_event = { id = DEMES.721 }
				any_vassal = {
					limit = {
						has_character_flag = attending_chariot_race
						NOT = { character = ROOT }
					}
					character_event = { id = DEMES.721 }
				}
			}
		}
		tooltip = {
			liege = {
				if = {
					limit = { NOT = { trait = drunkard } }
					opinion = {
						modifier = opinion_feast_scandal
						who = ROOT
					}
				}
				if = {
					limit = { 
						trait = drunkard 
						ROOT = { trait = green_deme }
						trait = green_deme
					}
					opinion = {
						modifier = opinion_drinking_buddy
						who = ROOT
					}
				}
				if = {
					limit = { 
						trait = drunkard 
						ROOT = { trait = blue_deme }
						trait = blue_deme
					}
					opinion = {
						modifier = opinion_drinking_buddy
						who = ROOT
					}
				}

				any_vassal = {
					limit = {
						has_character_flag = attending_chariot_race
						NOT = { character = ROOT }
						NOT = { trait = drunkard }
					}
					opinion = {
						modifier = opinion_feast_scandal
						who = ROOT
					}
				}
				any_vassal = {
					limit = {
						has_character_flag = attending_chariot_race
						NOT = { character = ROOT }
						trait = drunkard
					}
					opinion = {
						modifier = opinion_drinking_buddy
						who = ROOT
					}
				}
			}
		}
	}
}

# Drunkard (Host and Other Guests)
character_event = {
	id = DEMES.721
	desc = EVTDESC_DEMES_651
	picture = GFX_evt_drunk
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTA_TOG_651
		trigger = { 
			OR = {
				NOT = { trait = drunkard } 
				AND = {
					trait = drunkard 
					FROM = { trait = green_deme }
					trait = blue_deme				
				}
				
				AND = {
					trait = drunkard 
					FROM = { trait = blue_deme }
					trait = green_deme					
				}
			}
		}
		opinion = {
			modifier = opinion_feast_scandal
			who = FROM
		}
	}
	option = {
		name = EVTOPTB_TOG_651
		tooltip_info = drunkard
		trigger = { 
			OR = {
				AND = {
					trait = drunkard 
					FROM = { trait = blue_deme }
					trait = blue_deme
				}
				AND = {
					trait = drunkard 
					FROM = { trait = green_deme }
					trait = green_deme			
				}
			}
		}
		opinion = {
			modifier = opinion_drinking_buddy
			who = FROM
		}
	}
	option = {	# Start a Brawl
		name = EVTOPTC_TOG_651
		tooltip_info = drunkard
		trigger = {
			or = {
				has_character_flag = restless_race
				has_character_flag = rioting_race
			}
			OR = {
				AND = {
					trait = drunkard 
					FROM = { trait = blue_deme }
					trait = green_deme
				}
				AND = {
					trait = drunkard 
					FROM = { trait = green_deme }
					trait = blue_deme			
				}
				trait = wroth
			}
		}
		FROM = { character_event = { id = DEMES.670 } }
	}
}

character_event = {	#Start of Brawl
	id = DEMES.670
	desc = EVTDESC_DEMES.670
	picture = GFX_evt_drunk
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTA_DEMES.670
		trigger = {
			combat_rating_diff = {
				character = FROM
				value = 3
			}
		}
		top_liege = { character_event = { id = DEMES.201 } }
		character_event = { id = DEMES.671 }	#Win Brawl
		FROM = { character_event = { id = DEMES.672 } } #Lose Brawl
	}
	option = {
		name = EVTOPTA_DEMES.670
		trigger = {
			NOT = {
				combat_rating_diff = {
					character = FROM
					value = 3
				}
			}
		}
		top_liege = { character_event = { id = DEMES.201 } }
		random_list = {
			50 = {
				character_event = { id = DEMES.671 }	#Win Brawl
				FROM = { character_event = { id = DEMES.672 } } #Lose Brawl
			}
			50 = {
				FROM = { character_event = { id = DEMES.671 } }	#Win Brawl
				character_event = { id = DEMES.672 }  #Lose Brawl		
			}
		}
	}
}
