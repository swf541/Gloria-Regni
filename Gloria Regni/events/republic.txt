narrative_event = {
	id = commune.1
	title = "EVTNAMEcommune.1"
	desc = "EVTDESCcommune.1"
	picture = GFX_evt_roman_emperor_salute
	
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTAcommune.1"

	    liege = {
	 	k_roman_republic = {
			grant_title = ROOT
			copy_title_laws = d_commune_of_rome
		}
		set_defacto_vassal = ROOT
	    }

	    d_commune_of_rome = {
		destroy_landed_title = THIS
	    }
	    give_nickname = nick_the_glorious
	    hidden_tooltip = {
		activate_title = { title = d_commune_of_rome status = no }
	    }

	    any_playable_ruler = {
		limit = {
		    OR = {
			has_landed_title = e_hre
			has_landed_title = e_byzantium
			ai = no
		    }
		}
		    narrative_event = { id =  commune.99 }
	    }
	}
}

character_event = {
	id = commune.2
	desc = "EVTDESCcommune.2"
	picture = GFX_evt_roman_aquila_pillar
	
	trigger = {
		has_landed_title = k_roman_republic
		NOT = {
			has_landed_title = k_italy
			has_global_flag = goodbye_italy
		}
		completely_controls = d_ancona
		completely_controls = d_spoleto
		has_landed_title = c_roma
		completely_controls = c_orvieto
		completely_controls = c_orbetello
		completely_controls = d_toscana
	}
	
	mean_time_to_happen = {
		months = 1
		modifier = {
			factor = 0.5
			trait = ambitious
		}
		modifier = {
			factor = 2
			trait = content
		}
		modifier = {
			factor = 0.7
			trait = cynical
		}
		modifier = {
			factor = 0.4
			intrigue = 13
		}
		modifier = {
			factor = 0.9
			trait = gluttonous
		}
		modifier = {
			factor = 1.5
			learning = 13
		}
		modifier = {
			factor = 0.7
			prestige = 500
		}
		modifier = {
			factor = 0.6
			prestige = 750
		}
		modifier = {
			factor = 0.5
			prestige = 1000
		}
	}
	
	option = {
		name = "EVTOPTAcommune.2"
		prestige = 250
		if = {
			limit = {
				trait = content
			}
			remove_trait = content
		}
		set_global_flag = goodbye_italy
		hidden_tooltip = {
			activate_title = { title = d_commune_of_rome status = no }
			k_italy = {
				destroy_landed_title = THIS
			}
		}
		k_italy = {
			any_direct_de_jure_vassal_title = {
				de_jure_liege = k_roman_republic
			}
		}
	}
}
