	trigger = {
		age = 16
		is_female = no
		has_global_flag = colosseum
		NOT = {
			trait = craven
		}
	}
	
	mean_time_to_happen = {
		years = 15
		modifier = {
			factor = 0.1
			trait = duelist
		}
		modifier = {
			factor = 0.5
			trait = brave
		}
		modifier = {
			factor = 0.7
			trait = wroth
		}
		modifier = {
			factor = 0.7
			trait = tough_soldier
		}
		modifier = {
			factor = 0.8
			education = military
		}
		modifier = {
			factor = 1.5

		}

			factor = 1.7

		}
		modifier = {
			factor = 1.6
			trait = paranoid
		}
		modifier = {
			factor = 0.6
			trait = ambitious
		}
		modifier = {
			factor = 0.8
			trait = proud
		}
	}
	
	option = {
		name = "EVTOPTAgladiators.21"
		custom_tooltip = {
			text = try_a_fight
			hidden_tooltip = {
				if = {
					limit = {
						martial = 16
					}
					random_list = {
						30 = {
							character_event = { id = gladiators.22 }
						}
						20 = {
							if = {
								limit = {
									NOT = { trait = brave } 
								}
								add_trait = brave
							}
						}
						15 = {
							add_trait = wounded
						}
						10 = {
							add_trait = maimed
						}
						5 = {
							death = {
								death_reason = death_accident
							}
						}
					}
				}
				if = {
					limit = {
						NOT = {
							martial = 16
						}
					}
					random_list = {
						20 = {
							character_event = { id = gladiators.22 } 
						}
						25 = {
							if = {
								limit = {
									NOT = {
										trait = brave
									}
								}
								add_trait = brave
								add_trait = wounded
							}
						}
						20 = {
							add_trait = wounded
						}
						15 = {
							add_trait = maimed
						}
						10 = {
							death = {
								death_reason = death_accident
							}
						}
					}
				}
			}
		}
	}
	
	option = {
		name = "EVTOPTBgladiators.21"
		piety = 100
		prestige = -50
	}
}
