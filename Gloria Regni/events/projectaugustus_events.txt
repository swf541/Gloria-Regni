			dynasty = none

			d_saint_blaise = {

		}
		
		religion_authority = {
			modifier = religious_order_formed
			years = 10
		}
	}
	
	option = {
		name = OPTAPA_10027

			religion = miaphysite
		}
		
	}
	option = {
		name = OPTBPA_10027
