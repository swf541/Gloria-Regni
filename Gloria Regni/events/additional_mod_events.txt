# Fragmentation of Italian and Lombard Language
province_event = {
	id = PA.10330
	desc = EVTDESCPA.10330
	picture = GFX_evt_stone_church

	trigger = {
                year = 867
			OR = {
				AND = {
					culture = old_italian
					NOT = { county = { duchy = { title = d_latium } } }
				}
				AND = {
					culture = lombard
					NOT = { county = { duchy = { title = d_lombardia } } }
					NOT = { county = { duchy = { title = d_susa } } }
				}
			}
             NOT = { year = 1000 }
		OR = { 
			county = { kingdom = { title = k_lombardy } }
			county = { kingdom = { title = k_sicily } }
                }
	}
	
	mean_time_to_happen = {
		months = 2400
		modifier = {
			factor = 0.5
			owner = {
				culture_group = latin
			}
		}
	}

	option = {
		name = EVTOPTA_CM_2
		if = {
			limit = {
				county = { kingdom = { title = k_sicily } }
			}
			culture = beneventan
		}
		if = {
			limit = {
				OR = {
					county = { duchy = { title = d_spoleto } }
					county = { duchy = { title = d_ancona } }
				}	
			}
			culture = umbrian
		}
		if = {
			limit = {
				OR = {
					county = { duchy = { title = d_pisa } }
					county = { duchy = { title = d_toscana } }
					county = { duchy = { title = d_corsica } }
				}	
			}
			culture = tuscan
		}
		if = {
			limit = {
				OR = {
					county = { duchy = { title = d_sardinia } }
				}	
			}
			culture = sardinian
		}
		if = {
			limit = {
				OR = {
					county = { duchy = { title = d_genoa } }
					county = { duchy = { title = d_provence } }
				}	
			}
			culture = ligurian
		}
		if = {
			limit = {
				OR = {
					county = { duchy = { title = d_venice } }
					county = { duchy = { title = d_carinthia } }
					county = { duchy = { title = d_verona } }
				}	
			}
			culture = venetian
		}
		if = {
			limit = {
				county = { duchy = { title = d_ferrara } }
			}
			culture = romagnan
		}
	}
}
