namespace = PA

### The autokrator/Autokrateira Traits

# Augusta Trait Gained
character_event = {
	id = PA.10000
	desc = EVTDESCPA.10000
	picture = GFX_evt_rome_restored_byzantine
	
	trigger = {
		has_dlc = "Legacy of Rome"
        is_female = yes
		NOT = { trait = augusta }
		has_landed_title = e_roman_empire
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = OPTPA_10000
		add_trait = augusta
	}
}

# Augusta Trait Lost
character_event = {
	id = PA.10001
	desc = EVTDESCPA.10001
	picture = GFX_evt_rome_restored_byzantine
	
	trigger = {
		has_dlc = "Legacy of Rome"
		trait = augusta
		is_alive = yes
        OR = {
			NOT = { has_landed_title = e_roman_empire }
			}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = OPTPA_10001

	}
}
