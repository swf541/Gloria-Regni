
#------------------------------------------------------------------------------#
# 현재 스코프의 캐릭터가 고아원 보모인지 확인하는 조건문 트리거                #
#------------------------------------------------------------------------------#
# 사용 가능 스코프 : 고아원 보모인지 확인하기 위한 캐릭터                      #
#------------------------------------------------------------------------------#

is_orphanage_nurse_trigger = {
	OR = {
		has_minor_title = title_orphanage_nurse
		has_minor_title = title_orphanage_nurse_advisor
	}
}

#------------------------------------------------------------------------------#
# 고아원 보모가 수도에 있는지 확인하는 조건문 - 확인할 플래그/트레잇           #
#------------------------------------------------------------------------------#
# 사용 가능 스코프 : 고아원 보모 캐릭터                                        #
#------------------------------------------------------------------------------#

is_orphanage_nurse_in_capital_clause_trigger = {
	NOT = { has_character_flag = mod_character_event_flag_mission_to_battlefield }
	NOT = { has_character_flag = mod_character_event_flag_mission_to_rescue_orphan }
	NOT = { trait = on_a_journey }
}

#------------------------------------------------------------------------------#
# 고아원 보모가 수도에 있는지 확인하는 조건문 트리거                           #
#------------------------------------------------------------------------------#
# 사용 가능 스코프 : 플레이어                                                  #
#------------------------------------------------------------------------------#

is_orphanage_nurse_in_capital_trigger = {
	any_courtier = {
		is_orphanage_nurse_trigger = yes
		is_orphanage_nurse_in_capital_clause_trigger = yes
	}
}

#------------------------------------------------------------------------------#
# 고아들에 대한 관리 레벨이 높은 수준인지 확인                                 #
#------------------------------------------------------------------------------#
# 사용 가능 스코프 : 제한 없음                                                 #
#------------------------------------------------------------------------------#

is_orphan_management_level_high_trigger = {
	has_global_flag = mod_orphanage_policy_manage_level_high
}
