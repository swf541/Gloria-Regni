#==============================================================================#
# 가신, 자문회원 또는 배우자를 만나는 이벤트 (1201~)                           #
# 플레이어가 수도에 없는 경우, 현재 위치하는 지역 군주의 가신들을 만나게 된다. #

#------------------------------------------------------------------------------#
# 1201~1202 : 어떤 캐릭터를 만날지 선택 / 캐릭터 신분에 따른 이벤트 분기       #
#             만약 만날 수 있는 캐릭터가 없는 경우에는 1203 으로 분기한다.     #
#------------------------------------------------------------------------------#

character_event = {

	id = WLK.1201

	hide_window = yes
	is_triggered_only = yes

	# ROOT : Player

	immediate = {

		#----------------------------------------------------------------------#
		# 산책의 쿨 타임 (기본: 3일) 을 적용                                   #
		#----------------------------------------------------------------------#

		add_character_modifier = {
			name = mod_event_interval_notification_flag_modifier_go_on_a_walk
			duration = 3
		}

		#----------------------------------------------------------------------#
		# 현재 플레이어의 수도에 위치 중                                       #
		#----------------------------------------------------------------------#

		if = {
			limit = {
				is_player_in_capital_trigger = yes
			}

			#------------------------------------------------------------------#
			# 만날 수 있는 가신이 1명 이상 존재한다.                           #
			#------------------------------------------------------------------#

			if = {
				limit = {
					any_courtier = {
						count = 1
						is_common_adult_npc_trigger = yes
						at_location = ROOT
						NOT = { character = ROOT }
					}
				}

				random_courtier = {
					limit = {
						is_common_adult_npc_trigger = yes
						at_location = ROOT
						NOT = { character = ROOT }
					}

					save_event_target_as = mod_target_npc_to_meet
				}

				character_event = {
					id = WLK.1202
				}

				break = yes
			}

			#------------------------------------------------------------------#
			# 만날 수 있는 가신이 없다 : 보통은 발생하지 않지만...             #
			#------------------------------------------------------------------#

			character_event = {
				id = WLK.1203
			}
		}

		#----------------------------------------------------------------------#
		# 현재 플레이어가 타 지역에 위치 중 (주군의 자문회원 임무 등)          #
		#----------------------------------------------------------------------#

		if = {
			limit = {
				is_player_in_capital_trigger = no
			}

			#------------------------------------------------------------------#
			# 만날 수 있는 가신이 1명 이상 존재한다.                           #
			#------------------------------------------------------------------#

			if = {
				limit = {
					location = {
						holder_scope = {
							any_courtier = {
								count = 1
								is_common_adult_npc_trigger = yes
								at_location = ROOT
								NOT = { character = ROOT }
							}
						}
					}
				}

				location = {
					holder_scope = {
						random_courtier = {
							limit = {
								is_common_adult_npc_trigger = yes
								at_location = ROOT
								NOT = { character = ROOT }
							}

							save_event_target_as = mod_target_npc_to_meet
						}
					}
				}

				character_event = {
					id = WLK.1202
				}

				break = yes
			}

			#------------------------------------------------------------------#
			# 만날 수 있는 가신이 없다 : 보통은 발생하지 않지만...             #
			#------------------------------------------------------------------#

			character_event = {
				id = WLK.1203
			}
		}
	}
}

character_event = {

	id = WLK.1202

	is_triggered_only = yes
	hide_window = yes

	#--------------------------------------------------------------------------#
	# ROOT : Player                                                            #
	#                                                                          #
	# 만날 캐릭터는 WLK.1201에서 mod_target_npc_to_meet 이벤트 스코프에 넣었음 #
	#--------------------------------------------------------------------------#

	immediate = {

		#----------------------------------------------------------------------#
		# A. 나의 배우자 (배우자spouse 또는 첩consort) 를 만난 경우            #
		#    (지역에 관계 없이)                                                #
		#----------------------------------------------------------------------#

		if = {
			limit = {
				event_target:mod_target_npc_to_meet = {
					OR = {
						any_spouse = {
							character = ROOT
						}
						any_consort = {
							character = ROOT
						}
					}
				}
			}

			## 배우자 또는 첩을 만난 경우의 이벤트 : 항상 WLK.1210 으로

			event_target:mod_target_npc_to_meet = {
				character_event = {
					id = WLK.1210
				}
			}
			break = yes
		}

		#----------------------------------------------------------------------#
		# B. 자문회 구성원을 만난 경우                                         #
		#    (내 자문회 구성원인 경우에만 의미가 있다.)                        #
		#----------------------------------------------------------------------#

		if = {
			limit = {
				event_target:mod_target_npc_to_meet = {
					has_job_title = job_chancellor
				}
			}
			event_target:mod_target_npc_to_meet = {
				character_event = {
					id = WLK.1205
				}
			}
			break = yes
		}
		if = {
			limit = {
				event_target:mod_target_npc_to_meet = {
					has_job_title = job_marshal
				}
			}
			event_target:mod_target_npc_to_meet = {
				character_event = {
					id = WLK.1206
				}
			}
			break = yes
		}
		if = {
			limit = {
				event_target:mod_target_npc_to_meet = {
					has_job_title = job_treasurer
				}
			}
			event_target:mod_target_npc_to_meet = {
				character_event = {
					id = WLK.1207
				}
			}
			break = yes
		}
		if = {
			limit = {
				event_target:mod_target_npc_to_meet = {
					has_job_title = job_spymaster
				}
			}
			event_target:mod_target_npc_to_meet = {
				character_event = {
					id = WLK.1208
				}
			}
			break = yes
		}
		if = {
			limit = {
				event_target:mod_target_npc_to_meet = {
					has_job_title = job_spiritual
				}
			}
			event_target:mod_target_npc_to_meet = {
				character_event = {
					id = WLK.1209
				}
			}
			break = yes
		}

		#----------------------------------------------------------------------#
		# C. 그 외 가신courtier을 만난 경우                                    #
		#----------------------------------------------------------------------#

		event_target:mod_target_npc_to_meet = {
			character_event = {
				id = WLK.1204
			}
		}
	}
}
