#The Nehushtan Rediscovered
narrative_event = {
	id = 305
	desc = EVTDESC305
	picture = (Man holding Nehushtan head, two others looking at it with amazement)
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTA305
		if = {
			trait = cynical
			remove_trait = cynical
		}
		custom_tooltip = {
			text = rebuild_it
			hidden_tooltip = {
				if = {
					OR = {
						religion = rabbinic_judaism
						religion = temple_judaism
						religion = grand_sanhendrin
					}
					character_event = { id = 306 months = 10 }
				}
			}
			hidden_tooltip = {
				if = {
					NOT = {
						religion = rabbinic_judaism
						religion = temple_judaism
						religion = grand_sanhendrin
					}
					narrative_event = { id = 308 months = 24 }
				}
			}
		}
		prestige = 5000
		piety = 2500
	}
	
	option = {
		name = EVTOPTB305
		if = {
			trait = cynical
			remove_trait = cynical
		}
		custom_tooltip = {
			text = hide_this_artifact
		}
		prestige = 2500
		piety = 5000
	}
}
