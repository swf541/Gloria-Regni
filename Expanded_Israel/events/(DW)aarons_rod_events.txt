# A Lone Tabernacle
character_event = {
	id = 250
	desc = EVTDESC250
	picture = (3 Men riding to a Tabernacle in the middle of a Desert)

	min_age = 16
	prisoner = no
	capable_only = yes
	
	trigger = {
		has_dlc = "Sons of Abraham"
		religion_group = jewish_group
		has_landed_title = {
			OR = {
				kingdom = { title = k_jerusalem }
				kingdom = { title = k_israel }
			}
		}
		NOT = { has_global_flag = found_aarons_rod }
	}

	mean_time_to_happen = {
		months = 9600
		modifier = {
			factor = 2.0
			trait = zealous
		}
		modifier = {
			factor = 1.5
			trait = mystic
		}
		modifier = {
			factor = 2.0
			trait = theologian
		}
		modifier = {
			factor = 0.5
			trait = sympathy_christianity
		}
		modifier = {
			factor = 0.5
			trait = sympathy_islam
		}
		modifier = {
			factor = 0.5
			trait = sympathy_zoroastrian
		}
		modifier = {
			factor = 0.5
			trait = sympathy_indian
		}
		modifier = {
			factor = 0.25
			trait = sympathy_pagan
		}
		modifier = {
			factor = 0.1
			trait = cynical
		}
	}
	
	option = {
		name = EVTOPTA250
		custom_tooltip = {
			text = go_to_it
			hidden_tooltip = {
				random_list = {
					25 = { character_event = { id = 251 days = 1 } }
					25 = { character_event = { id = 252 days = 1 } }
					50 = { character_event = { id = 253 days = 1 } }
				}
			}
		}
	}
}
