narrative_event = {
	id = JMI.1
	title = "Jewish Settlements In England"
	desc = EVTDESCLoR.210
	
	picture = GFX_evt_icon

	trigger = {
		has_law = full_rights
	}


	mean_time_to_happen = {
		months = 240
				
	}

	option = { # Yes
	name = EVTOPTALoR.211

	any_realm_province = {
		limit = {
			has_province_modifier = rabbinic_minority
		}

		random_neighbor_province = {
			limit = {
				not = { has_province_modifier = rabbinic_minority }
			}
			add_province_modifier = rabbinic_minority
			}
		}
	}
}
